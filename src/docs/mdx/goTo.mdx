import Carousel from 'react-elastic-carousel';

# goTo

You can trigger the `goTo` method via a `ref` to the `Carousel`.

<Playground>
    {
        function App() {
            const carouselRef = React.useRef(null);
    
            function handleGoto({target: {value}){
                carouselRef.current.goTo(Number(value))
            }
    
            return(
                <div>
                    <label>
                        Go to 
                        <input type="number" defaultValue={0} onChange={handleGoto} />
                    </label>
                    
                    <hr/>
                    
                    <Carousel ref={carouselRef}>
                        <Item>1</Item>
                        <Item>2</Item>
                        <Item>3</Item>
                        <Item>4</Item>
                        <Item>5</Item>
                        <Item>6</Item>
                    </Carousel>
                </div>
            );
        }
    }

</Playground>
